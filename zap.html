// Código para correr no Firebase Cloud Functions (Node.js) const functions = require('firebase-functions'); const admin = require('firebase-admin'); // Inicializa o acesso de Administrador à base de dados admin.initializeApp(); exports.enviarNotificacaoChat = functions.firestore .document('chat_equipe/{mensagemId}') .onCreate(async (snap, context) => { const mensagem = snap.data(); const senderId = mensagem.senderId; const receiverId = mensagem.receiverId; // Se não houver destinatário específico, não enviamos notificação if (!receiverId) return null; try { // 1. Obter o nome de quem enviou a mensagem const senderDoc = await admin.firestore().collection('usuarios').doc(senderId).get(); const senderName = senderDoc.exists ? senderDoc.data().nome : 'Colega da Equipa'; // 2. Obter os dados de quem vai receber a mensagem (para apanhar o Token do telemóvel) const receiverDoc = await admin.firestore().collection('usuarios').doc(receiverId).get(); if (!receiverDoc.exists) return null; const receiverData = receiverDoc.data(); const fcmToken = receiverData.fcmToken; // Se o destinatário ainda não aceitou as notificações (não tem token), cancelamos if (!fcmToken) { console.log(`O utilizador ${receiverId} não tem um fcmToken configurado.`); return null; } // 3. Montar o texto do balão de notificação // Se for áudio, diz "Mensagem de Áudio", senão mostra o texto const textoNotificacao = mensagem.audioData ? "🎵 Mensagem de Áudio" : mensagem.text; // 4. Preparar o pacote a ser enviado para o telemóvel const payload = { notification: { title: `WG: Mensagem de ${senderName}`, body: textoNotificacao, }, token: fcmToken }; // 5. Disparar a notificação oficial do Google (que acorda o telemóvel em segundo plano) const response = await admin.messaging().send(payload); console.log('Notificação PUSH enviada com sucesso:', response); return null; } catch (error) { console.error('Falha crítica ao tentar enviar notificação:', error); return null; } });
